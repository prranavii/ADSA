## leetcode 441. Arranging Coins

https://leetcode.com/submissions/detail/1810393228/

```java
class Solution {
    public int arrangeCoins(int n) {
        int start = 0;
        int end = n;
        int ans = 0; 

        while (start <= end) {
            long mid = start + (end - start) / 2;
          
            long total = mid * (mid + 1) / 2;

            if (total == n) {
                return (int)mid;
            } else if (total < n) {
                ans = (int)mid; 
                start = (int)mid + 1;
            } else {
                end = (int)mid - 1; 
            }
        }

        return ans; 
    }
}
