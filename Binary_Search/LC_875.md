## KOKO EATING BANANAS

https://leetcode.com/submissions/detail/1786463400/

```java
class Solution {
    public int minEatingSpeed(int[] piles, int h) {
        int max = 0;
        for (int pile : piles) {
            if (pile > max) max = pile;
        }

        int left = 1;
        int right = max;

        while (left < right) {
            int mid = (left + right) / 2;
            int hoursNeeded = 0;

            for (int pile : piles) {
                hoursNeeded += (pile + mid - 1) / mid;
            }

            if (hoursNeeded <= h) {
            
                right = mid;
            } else {
            
                left = mid + 1;
            }
        }

        return left;
    }
}
