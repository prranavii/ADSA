## leetcode 1052. Grumpy Bookstore Owner

https://leetcode.com/submissions/detail/1793234992/

```java

class Solution {
    public int maxSatisfied(int[] customers, int[] grumpy, int minutes) {
        int n = customers.length;
        int baseHappy = 0; 
        for (int i = 0; i < n; i++) {
            if (grumpy[i] == 0) {
                baseHappy += customers[i];
            }
        }

        int extraHappy = 0;      
        int maxExtraHappy = 0;   
        for (int i = 0; i < n; i++) {
        
            if (grumpy[i] == 1) {
                extraHappy += customers[i];
            }

          
            if (i >= minutes && grumpy[i - minutes] == 1) {
                extraHappy -= customers[i - minutes];
            }

            maxExtraHappy = Math.max(maxExtraHappy, extraHappy);
        }

        return baseHappy + maxExtraHappy;
    }
}
