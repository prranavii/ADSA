## leetcode 1004 Max Consecutive Ones III

https://leetcode.com/submissions/detail/1793200556/

```java

class Solution {
    public int longestOnes(int[] nums, int k) {
        int left=0;
        int len=0;
        int zeroes=0;

        for(int right=0; right<nums.length;right++){
            if(nums[right]==0){
                zeroes++;
            }
        
        while(zeroes>k){
            if(nums[left]==0){
                zeroes--;
            }
            left++;
        }
        

        len=Math.max(len, right-left+1);
    
    }
     return len;
   
}
 }
